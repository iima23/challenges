abstract class Poupee {
    private int size;
    private boolean open;

    public Poupee(int size) {
        this.size = size;
    }

    public int getSize() {
        return size;
    }

    public boolean isOpen() {
        return open;
    }

    public void setSize(int size) {
        this.size = size;
    }

    public void setOpen(boolean open) {
        this.open = open;
    }

    public abstract void openPoupee();

    public abstract void closePoupee();

    public abstract void placeIn(Poupee p);

    public abstract void takeOut(Poupee p);

}

import java.util.Scanner;                                 
class Menu {                                              
    public  PoupeeRusse poupee1;                          
    public  PoupeeRusse poupee2;                          
                                                          
    public Menu(PoupeeRusse poupee1, PoupeeRusse poupee2) 
        this.poupee1 = poupee1;                           
        this.poupee2 = poupee2;                           
    }                                                     
                                                          
    public void displayMenu() {                           
        Scanner scanner = new Scanner(System.in);         
        int option;                                       
        do {                                              
            System.out.println("Menu ");                  
            System.out.println("1. Open Poupee");         
            System.out.println("2. Close Poupee");        
            System.out.println("3. PlaceIn(Poupee p)");   
            System.out.println("4. Takeout(Poupee p)");   
            System.out.println("5. exit");                
                                                          
            option = scanner.nextInt();                   
                                                          
            switch (option) {                             
                case 1:                                   
                    poupee1.openPoupee();                 
                    break;                                
                case 2:                                   
                    poupee1.closePoupee();                
                    break;                                
                case 3:                                   
                    poupee1.placeIn(poupee2);             
                    break;                                
                case 4:                                   
                    poupee1.takeOut(poupee2);             
                    break;                                
                case 5:                                   
                    System.out.println("Exit");           
                    break;                                
                default:                                  
                    System.out.println("Invalid option"); 
                    break;                                
            }                                             
        } while (option != 5);                            
        scanner.close();                                  
    }                                                     
}
